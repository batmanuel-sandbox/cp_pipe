166 class CppTask(pipeBase.CmdLineTask):
167  """
168  Calibration Products Production (CPP) task.
169 
170  This task is used to produce the calibration products required to calibrate cameras.
171  Examples of such operations are as follows:
172  Given a set of flat-field images, find the dark pixels and columns.
173  Given a set of darks, find the bright pixels and columns.
174  Given a set of Fe55 exposures, calulate the gain of the readout chain, in e-/ADU
175  Given a set of Fe55 exposures, calulate the instrinsic PSF of the silicon, and the degradation of
176  the PSF due to CTE.
177  Given a set of flat-pairs, measure the photon transfer curve (PTC).
178  Given a set of bias frames, calculate the read noise of the system in e-.
179  Given a set of pocket-pumping exposures, find charge-traps in the silicon.
180 
181  The CppTask.runEotestDirect() is only applicable to LSST sensors, and only for a specific type of dataset
182  This method takes a dafPersistance.Butler corresponding to a repository in which a full eotest run has
183  been taken and ingested, and runs each of the tasks in eotest directly, allowing for bitwise comparison
184  with results given by the camera team.
185 
186  See http://ls.st/ldm-151 Chapter 4, Calibration Products Production for further details
187  regarding the inputs and outputs.
188  """

228  def _getMaskFiles(self, path, ccd):
229  """Get all available eotest mask files for a given ccd.
230 
231  Each stage of the processing generates more mask files, so this allows each to be picked up
232  as more and more tests run, and saves having to have clever logic for if some tasks fail.
233  Parameters
234  ----------
235  path : string
236  Path on which to find the mask files
237  ccd : string
238  Name/identifier of the CCD
239  Returns
240  -------
241  maskFiles : list/tuple
242  List of mask files, or an empty tuple if none are found
243  """
244  pattern = '*' + ccd + '*mask*'
245  maskFiles = glob.glob(os.path.join(path, pattern))
246  return maskFiles if len(maskFiles) > 0 else () # eotest wants an empty tuple here

260  def _gainPropSetToDict(self, pSet):
261  """Translator for the persisted gain values.
262 
263  eotest wants an {amp: gain} dictionary with integer keys for the amps.
264  When we persist dafBase.propertySets these only take strings for keys,
265  so the provided .toDict() method won't do, so here we provide an extra
266  layer of translation to keep eotest happy.
267 
268  Parameters
269  ----------
270  pSet : daf.base.PropertySet
271  PropertySet to be translated to an integer-keyed dictionary
272  """
273  return {int(amp): gain for amp, gain in pSet.toDict().items()}

281  def runEotestDirect(self, butler, run=None):
282  """
283  Generate calibration products using eotest algorithms.
284 
285  Generate all calibration products possible using the vanilla eotest implementation,
286  given a butler for a TS8 (raft-test) repo. It can contain multiple runs, but must correspond to
287  only a single raft/RTM.
288 
289  - Run all eotest tasks possible, using the butler to gather the data
290  - Write outputs in eotest format
291 
292  In order to replicate the canonical eotest analysis, the tasks should be run in a specific order.
293  This is given/defined in the "Steps" section here:
294  http://lsst-camera.slac.stanford.edu/eTraveler/exp/LSST-CAMERA/
295  displayProcess.jsp?processPath=1179
296 
297  But is replicated here for conveniece:
298  * 55Fe Analysis
299  * CCD Read Noise Analysis
300  * Bright Defects Analysis
301  * Dark Defects Analysis
302  * Traps Finding
303  * Dark Current X - will not be implemented here
304  * Charge Transfer Efficiencies
305  * Photo-response analysis X - will not be implemented here
306  * Flat Pairs Analysis
307  * Photon Transfer Curve
308  * Quantum Efficiency X - will not be implemented here
309 
310  List of tasks that exist in the eotest package but aren't mentioned on the above link:
311  --------
312  # linearityTask()
313  # fe55CteTask()
314  # eperTask()
315  # crosstalkTask()
316  # persistenceTask()
317 
318  # TODO: For each eotest task, find out what the standard raft testing does for the optional params.
319  i.e. many have optional params for gains, bias-frames etc - if we want bitwise identicallity then we
320  need to know what is typically provided to these tasks when the camera team runs this code.
321  This can probably be worked out from https://github.com/lsst-camera-dh/lcatr-harness
322  but it sounds like Jim Chiang doesn't recommend trying to do that.
